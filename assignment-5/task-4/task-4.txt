public class Animal{
    public int legs = 4;
    public String sound = "Not defined";
    public void details(){
        System.out.println("Legs: "+legs);
        System.out.println("Sound: "+sound);
    }
}


public class Dog extends Animal{
    public String name;

    @Override
    public void details() {
        super.details();
    }
    public String getName(){
        return name;
    }
    public void updateSound(String a){
        sound=a;
    }
}
