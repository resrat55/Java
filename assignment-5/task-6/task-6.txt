public class Book extends Product{
    public String isbn;
    public String publishar;
    public Book(int a,String b,int c,String d,String e){
        super(a,b,c);
        isbn=d;
        publishar=e;

    }
    public String  printDetail(){
        return getIdTitlePrice()+"\nISBN: "+isbn+" Publisher:"+publishar;
    }
}


public class CD extends Product{
    public String band;
    public int duration;
    public String genre;
    public CD(int a,String b,int c,String d,int e,String f){
        super(a,b,c);
        band=d;
        duration=e;
        genre=f;

    }
    public String printDetail(){
        return getIdTitlePrice()+"\nBand: "+band+" Duration: "+duration+" minutes\nGenre: "+genre;
    }
}
